import Image from 'next/image';

import images from '../../data/images.json';

# Next Image

Grid of images using the Next Image component.

export const Grid = () => {
  return (
    <ul className="grid grid-cols-2 lg:grid-cols-3 gap-6 my-8">
      {images.map((image) => {
        return (
          <li key={image.src}>
            <Image
              src={image.src}
              width={image.width}
              height={image.height}
              alt={image.title}
            />
          </li>
        );
      })}
    </ul>
  )
}
